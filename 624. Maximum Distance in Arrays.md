#624. Maximum Distance in Arrays

找两个数的绝对值的最大的距离，memory beats 100%

    def maxDistance(self, arrays):
        """
        :type arrays: List[List[int]]
        :rtype: int
        """
        
        Max, Min = -float('inf'), float('inf')
        result = 0
        for line in arrays:
            result = max(Max - line[0], result)
            result = max(line[-1]-Min, result)
            Max = max(Max, line[-1])
            Min = min(Min, line[0])
            
        return result
